        // State
        let currentScripts = [];
        let currentSearchQuery = "";
        let isLoading = false;

        // DOM Elements
        const searchForm = document.getElementById('searchForm');
        const searchInput = document.getElementById('searchInput');
        const refreshBtn = document.getElementById('refreshBtn');
        const searchInfo = document.getElementById('searchInfo');
        const searchQuery = document.getElementById('searchQuery');
        const resultsCount = document.getElementById('resultsCount');
        const errorAlert = document.getElementById('errorAlert');
        const errorMessage = document.getElementById('errorMessage');
        const loadingState = document.getElementById('loadingState');
        const loadingText = document.getElementById('loadingText');
        const scriptsGrid = document.getElementById('scriptsGrid');
        const noResults = document.getElementById('noResults');
        const noResultsMessage = document.getElementById('noResultsMessage');
        const modalOverlay = document.getElementById('modalOverlay');
        const modalTitle = document.getElementById('modalTitle');
        const modalGame = document.getElementById('modalGame');
        const modalPatchedBadge = document.getElementById('modalPatchedBadge');
        const modalStats = document.getElementById('modalStats');
        const modalTags = document.getElementById('modalTags');
        const modalDescription = document.getElementById('modalDescription');
        const scriptCode = document.getElementById('scriptCode');
        const copyBtn = document.getElementById('copyBtn');
        const modalClose = document.getElementById('modalClose');

        // Utility Functions
        function formatNumber(num) {
            if (num >= 1000000) return `${(num / 1000000).toFixed(1)}M`;
            if (num >= 1000) return `${(num / 1000).toFixed(1)}K`;
            return num.toString();
        }

        function formatDate(dateString) {
            try {
                return new Date(dateString).toLocaleDateString('en-US', {
                    year: 'numeric',
                    month: 'long',
                    day: 'numeric'
                });
            } catch {
                return 'Unknown';
            }
        }

        function delay(ms) {
            return new Promise(resolve => setTimeout(resolve, ms));
        }

        // Mock API
        const scriptBloxAPI = {
            async getScripts(page = 1, limit = 20) {
                await delay(800);
                const startIndex = (page - 1) * limit;
                const endIndex = startIndex + limit;
                return mockScripts.slice(startIndex, endIndex);
            },

            async searchScripts(query, page = 1, limit = 20) {
                await delay(600);
                
                if (!query.trim()) {
                    return this.getScripts(page, limit);
                }

                const filtered = mockScripts.filter(script => 
                    script.title.toLowerCase().includes(query.toLowerCase()) ||
                    script.description.toLowerCase().includes(query.toLowerCase()) ||
                    script.game?.toLowerCase().includes(query.toLowerCase()) ||
                    script.author.toLowerCase().includes(query.toLowerCase()) ||
                    script.tags.some(tag => tag.toLowerCase().includes(query.toLowerCase()))
                );

                const startIndex = (page - 1) * limit;
                const endIndex = startIndex + limit;
                return filtered.slice(startIndex, endIndex);
            }
        };

        // UI Functions
        function showElement(element) {
            element.classList.remove('hidden');
        }

        function hideElement(element) {
            element.classList.add('hidden');
        }

        function setLoading(loading) {
            isLoading = loading;
            if (loading) {
                showElement(loadingState);
                refreshBtn.classList.add('spinning');
                hideElement(scriptsGrid);
                hideElement(noResults);
                hideElement(errorAlert);
            } else {
                hideElement(loadingState);
                refreshBtn.classList.remove('spinning');
            }
        }

        function showError(message) {
            errorMessage.textContent = message;
            showElement(errorAlert);
            hideElement(scriptsGrid);
            hideElement(noResults);
            hideElement(loadingState);
        }

        function updateSearchInfo(query, count) {
            if (query) {
                searchQuery.textContent = query;
                resultsCount.textContent = `${count} results`;
                showElement(searchInfo);
            } else {
                hideElement(searchInfo);
            }
        }

        function createScriptCard(script) {
            const card = document.createElement('div');
            card.className = 'script-card';
            card.onclick = () => openModal(script);

            const patchedBadge = script.isPatched ? 
                'Patched' : '';

            const tagsHtml = script.tags.slice(0, 3).map(tag => 
                `${tag}`
            ).join('');

            const moreTags = script.tags.length > 3 ? 
                `+${script.tags.length - 3} more` : '';

            card.innerHTML = `
                

                    

                        
${script.title}

                        ${script.game ? `
${script.game}
` : ''}
                    

                    ${patchedBadge}
                

                
                
${script.description || 'No description available'}

                
                

                    ${tagsHtml}
                    ${moreTags}
                

                
                

                    

                        

                            
                            ${formatNumber(script.views)}
                        

                        

                            
                            ${script.author}
                        

                        

                            
                            ${new Date(script.created).toLocaleDateString()}
                        

                    

                    
                    
                

            `;

            return card;
        }

        function renderScripts(scripts) {
            scriptsGrid.innerHTML = '';
            
            if (scripts.length === 0) {
                showElement(noResults);
                noResultsMessage.textContent = currentSearchQuery 
                    ? `No scripts found matching "${currentSearchQuery}". Try a different search term.`
                    : "No scripts are currently available. Try refreshing the page.";
                hideElement(scriptsGrid);
                return;
            }

            hideElement(noResults);
            showElement(scriptsGrid);
            
            scripts.forEach(script => {
                const card = createScriptCard(script);
                scriptsGrid.appendChild(card);
            });
        }

        function openModal(script) {
            modalTitle.textContent = script.title;
            modalGame.textContent = script.game || '';
            modalDescription.textContent = script.description;
            scriptCode.textContent = script.scriptContent;

            if (script.isPatched) {
                showElement(modalPatchedBadge);
            } else {
                hideElement(modalPatchedBadge);
            }

            // Stats
            modalStats.innerHTML = `
                

                    
                    ${script.author}
                

                

                    
                    ${formatNumber(script.views)} views
                

                

                    
                    ${formatNumber(script.likes)} likes
                

                

                    
                    ${formatDate(script.created)}
                

            `;

            // Tags
            modalTags.innerHTML = script.tags.map(tag => 
                `${tag}`
            ).join('');

            modalOverlay.classList.add('open');
            document.body.style.overflow = 'hidden';
        }

        function closeModal() {
            modalOverlay.classList.remove('open');
            document.body.style.overflow = '';
            
            // Reset copy button
            copyBtn.innerHTML = `
                
                Copy Script
            `;
            copyBtn.classList.remove('copied');
        }

        async function copyToClipboard() {
            try {
                await navigator.clipboard.writeText(scriptCode.textContent);
                copyBtn.innerHTML = `
                    
                    Copied!
                `;
                copyBtn.classList.add('copied');
                
                // Show toast notification
                showToast('Script copied to clipboard!', 'success');
                
                setTimeout(() => {
                    if (copyBtn.classList.contains('copied')) {
                        copyBtn.innerHTML = `
                            
                            Copy Script
                        `;
                        copyBtn.classList.remove('copied');
                    }
                }, 2000);
            } catch (error) {
                showToast('Failed to copy to clipboard', 'error');
            }
        }

        function showToast(message, type = 'info') {
            // Simple toast implementation
            const toast = document.createElement('div');
            toast.style.cssText = `
                position: fixed;
                top: 20px;
                right: 20px;
                background: ${type === 'error' ? 'hsl(0, 84%, 60%)' : type === 'success' ? 'hsl(142, 71%, 45%)' : 'hsl(213, 94%, 68%)'};
                color: white;
                padding: 12px 16px;
                border-radius: 8px;
                font-size: 14px;
                font-weight: 500;
                z-index: 1000;
                animation: slideIn 0.3s ease-out;
            `;
            
            const style = document.createElement('style');
            style.textContent = `
                @keyframes slideIn {
                    from { transform: translateX(100%); opacity: 0; }
                    to { transform: translateX(0); opacity: 1; }
                }
                @keyframes slideOut {
                    from { transform: translateX(0); opacity: 1; }
                    to { transform: translateX(100%); opacity: 0; }
                }
            `;
            document.head.appendChild(style);
            
            toast.textContent = message;
            document.body.appendChild(toast);
            
            setTimeout(() => {
                toast.style.animation = 'slideOut 0.3s ease-in';
                setTimeout(() => {
                    document.body.removeChild(toast);
                    document.head.removeChild(style);
                }, 300);
            }, 3000);
        }

        // Main Functions
        async function loadScripts() {
            try {
                setLoading(true);
                hideElement(errorAlert);
                const data = await scriptBloxAPI.getScripts();
                currentScripts = data;
                renderScripts(data);
                updateSearchInfo(currentSearchQuery, data.length);
            } catch (error) {
                showError("Failed to load scripts. Please try again.");
                console.error("Error loading scripts:", error);
            } finally {
                setLoading(false);
            }
        }

        async function handleSearch(query) {
            try {
                setLoading(true);
                hideElement(errorAlert);
                currentSearchQuery = query;
                loadingText.textContent = query ? "Searching scripts..." : "Loading scripts...";
                
                const data = query.trim() 
                    ? await scriptBloxAPI.searchScripts(query)
                    : await scriptBloxAPI.getScripts();
                
                currentScripts = data;
                renderScripts(data);
                updateSearchInfo(query, data.length);
            } catch (error) {
                showError("Search failed. Please try again.");
                console.error("Error searching scripts:", error);
            } finally {
                setLoading(false);
            }
        }

        function handleRefresh() {
            if (currentSearchQuery) {
                handleSearch(currentSearchQuery);
            } else {
                loadScripts();
            }
        }

        // Event Listeners
        searchForm.addEventListener('submit', (e) => {
            e.preventDefault();
            handleSearch(searchInput.value);
        });

        searchInput.addEventListener('input', (e) => {
            const value = e.target.value;
            if (value.length === 0) {
                handleSearch("");
            }
        });

        refreshBtn.addEventListener('click', handleRefresh);
        modalClose.addEventListener('click', closeModal);
        copyBtn.addEventListener('click', copyToClipboard);

        modalOverlay.addEventListener('click', (e) => {
            if (e.target === modalOverlay) {
                closeModal();
            }
        });

        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape' && modalOverlay.classList.contains('open')) {
                closeModal();
            }
        });

        // Initialize
        document.addEventListener('DOMContentLoaded', () => {
            loadScripts();
        });

        // Make openModal global for onclick handlers
        window.openModal = openModal;
